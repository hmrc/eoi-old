Private Sub cmdCalculate_Click()
  
    SPPRecoverableCalculation "tblSPPAdoptionPayments", "tblSPPAdoptionRecoveryRates", Me
End Sub

Private Sub cmdClear_Click()
    
    On Error GoTo Err_cmdClear_Click
    
    DoCmd.SetWarnings False
    
    'Runs the required SQL to delete all of the data input on the subforms
    DoCmd.RunSQL "DELETE * FROM [tblSPPAdoptionPayments]"
    
    ' Update Details section, no records
    Me![subPayments].Form.Requery

    'Re-initialises all of the input fields and set focus back to first input field
    
    ' Clear information only fields
    txtName = Null
    txtInitials = Null
    NIPrefix = Null
    NInumber = Null
    NIsuffix = Null
    txtRef = Null
    

    ' Set focus
    txtName.SetFocus
    
    DoCmd.SetWarnings True
Exit_cmdClear_Click:
    Exit Sub

Err_cmdClear_Click:
    MsgBox Error$
    Resume Exit_cmdClear_Click

End Sub
